local u149 = {}

function updatePlayers()
    u149 = {}

    local v150 = next
    local v151, v152 = game.Players:GetPlayers()

    while true do
        local v153

        v152, v153 = v150(v151, v152)

        if v152 == nil then
            break
        end

        u149[v152] = v153.Name
    end
end
